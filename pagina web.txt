<script>
  function mostrarSeccion(id) {
    // Ocultar todas las secciones
    var secciones = document.querySelectorAll('.seccion');
    secciones.forEach(function(seccion) {
      seccion.style.display = 'none';
    });

    // Mostrar la sección seleccionada
    var seccionSeleccionada = document.getElementById(id);
    if (seccionSeleccionada) {
      seccionSeleccionada.style.display = 'block';
    }
  }

  function mostrarCrearUsuario() {
    document.getElementById('formInicioSesion').style.display = 'none';
    document.getElementById('formCrearUsuario').style.display = 'block';
  }

  function crearUsuario() {
    var nombreUsuario = document.getElementById('nombreUsuario').value;
    var contrasena = document.getElementById('contrasena').value;

    if (nombreUsuario && contrasena) {
      alert('Usuario creado correctamente.');
      mostrarSeccion('inicio');
    } else {
      alert('Por favor, complete todos los campos.');
    }
  }

  function iniciarSesion() {
    var nombreUsuario = document.getElementById('nombreUsuarioInicio').value;
    var contrasena = document.getElementById('contrasenaInicio').value;

    if (nombreUsuario === 'usuario' && contrasena === 'contraseña') {
      alert('Inicio de sesión exitoso.');
      mostrarSeccion('menuPrincipal');
    } else {
      alert('Nombre de usuario o contraseña incorrectos.');
    }
  }

  function buscarSoftware() {
    var query = document.getElementById('busqueda').value;
    alert('Buscando software: ' + query);
  }

  function buscarRecursos() {
    var query = document.getElementById('busquedaCapacitacion').value;
    alert('Buscando recursos: ' + query);
  }

  function comprobarVersion() {
    alert('Comprobando la versión del sistema...');
  }

  function mostrarVersionActual() {
    document.getElementById('infoVersion').style.display = 'block';
  }

  function actualizarSistema() {
    alert('Actualizando el sistema...');
  }

  function informarProblema() {
    alert('Por favor, describa el problema que está experimentando.');
  }

  }
